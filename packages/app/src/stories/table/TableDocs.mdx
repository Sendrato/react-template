## Example

```jsx
import React from 'react';
import GenericTable from '@modules/common/Table/GenericTable';
import { useSort, usePagination } from 'hooks';

const Table = () => {
  const sortConfig = useSort();
  const pagiantionConfig = usePagination();

  const config = [
    { key: 'id', label: 'Id' },
    { key: 'name', label: 'Name' },
    { key: 'age', label: 'Age' },
  ];

  const data = [
    { id: 1, name: 'John Doe', age: 25 },
    { id: 2, name: 'Jane Smith', age: 30 },
    { id: 3, name: 'Bob Johnson', age: 35 },
  ];

  return (
    <ReactTable
      {...sortConfig}
      {...pagiantionConfig}
      totalPage={100}
      data={data}
      config={config}
      title="Sample React Table"
    />
  );
};

export default Table;
```
